ğŸ“‹ Database Schema

Schema Definitions (shared/schema.ts)

// Destinations Table
export const destinations = pgTable("destinations", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  name: text("name").notNull(),
  day: text("day").notNull(),
  type: text("type").notNull(),
  location: text("location").notNull(),
  cost: integer("cost").notNull(),
  description: text("description").notNull(),
  latitude: real("latitude").notNull(),
  longitude: real("longitude").notNull(),
});
// Expenses Table (Persistent Storage)
export const expenses = pgTable("expenses", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  name: text("name").notNull(),
  type: text("type").notNull(),
  amount: integer("amount").notNull(),
  paidBy: text("paidBy").notNull().default("split"),
  expenseDate: text("expenseDate").notNull(),
  createdAt: bigint("createdAt", { mode: "number" }).notNull(),
});
// Messages Table
export const messages = pgTable("messages", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  sender: text("sender").notNull(),
  text: text("text").notNull(),
  timestamp: bigint("timestamp", { mode: "number" }).notNull(),
});
// Photos Table
export const photos = pgTable("photos", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  url: text("url").notNull(),
  caption: text("caption").notNull(),
  date: integer("date").notNull(),
  width: integer("width"),
  height: integer("height"),
});
// Users Table
export const users = pgTable("users", {
  id: varchar("id").primaryKey().default(sql`gen_random_uuid()`),
  username: text("username").notNull().unique(),
  password: text("password").notNull(),
});

ğŸ”Œ API Endpoints

Destination Endpoints

GET /api/destinations

Fetch all McLeodganj destinations with filters

GET /api/destinations/:id

Get a single destination by ID

GET /api/visited

Get all visited destinations

Expense Endpoints

GET /api/expenses

Fetch all expenses (persistent storage)

POST /api/expenses

Add new expense | Body: { name, type, amount, paidBy, expenseDate }

PUT /api/expenses/:id

Update expense | Body: { name?, type?, amount?, paidBy?, expenseDate? }

DELETE /api/expenses/:id

Delete expense by ID

Message Endpoints

GET /api/messages

Fetch all messages between Reh & Sim

POST /api/messages

Send message | Body: { sender, text, timestamp }

DELETE /api/messages

Delete all messages (batch clear)

Photo Endpoints

GET /api/photos

Fetch all trip photos

POST /api/photos

Add photo | Body: { url, caption, date, width?, height? }

DELETE /api/photos/:id

Delete photo by ID
